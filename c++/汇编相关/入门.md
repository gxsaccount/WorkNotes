## 常见特殊指令的替代 ##  

注：eip不能直接由程序员修改  
原指令|替换指令  
-|-
pushl %eax | subl $4,%esp; movl %eax,(%esp)  
popl %eax | movl (%esp),%eax;addl $4,%esp 
call 0x12345 | pushl %(eip),mov $0x12345,%eip
ret | popl %eip
