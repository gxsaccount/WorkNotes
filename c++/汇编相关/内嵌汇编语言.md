    c语言下文
    
 __asm__可以由asm代替  
 
 实例：    
    asm volatile( //
        "movl $0,%%eax\n\t"  //clear %eax to 0;%转转义字符  
        "addl %1,%%eax\n\t"   //%eax+=val1，%1指val1 
        "addl %2,%%eax\n\t"   //%eax+=val2
        "movl %%eax,%0/n/t"   //val2=%eax  
        :"=m"(val3)           //“=m”代表写道内存变量（非寄存器。m代表memory） 
        :"c"(val1),"d"(val2)  //输入的c或者d指%ecx和%edx，用%ecx存val1的值...
    )
